> 算法题之「两数之和」

给定一个整数数组和一个目标值，找出数组中和为目标值的两个数。

你可以假设每个输入只对应一种答案，且同样的元素不能被重复利用。

示例：

```javascript
给定 nums = [2, 7, 11, 15], target = 9

因为 nums[0] + nums[1] = 2 + 7 = 9
所以返回 [0, 1]
```

答案：

```javascript

let nums = [2, 7, 11, 15], target = 9

// 法一：两层for循环正向比较
function count(nums, target) {
    let len = nums.length

    let arr = []
    let subarr = []
    for (let i = 0; i < len; i++) {
        for (let j = i + 1; j < len; j++) {
            if (nums[i] + nums[j] === target) {
                arr.push(nums[i], nums[j])
                subarr.push(i, j)
                break;
            } 
        }
    }
    if (arr === []) {
        return 'no result'
    } else {
        console.log('数值：' + arr)
        return subarr
    }

}
// 法二：for循环没有嵌套
function anwser (arr, target) {
  let map = {}
  for (let i = 0; i < arr.length; i++) {
    map[arr[i]] = i
  }
  for (let i = 0; i < arr.length; i++) {
    let d = target - arr[i]
    if (map[d] && map[d] != i) {
      return [i, map[d]]
    }
  }
  return new Error('no result')
}
```
